{{- /* Markdown image render hook. Uses the central image.html partial. - Alt text (.Text) becomes the alt attribute - Title (.Title) becomes the visible caption - Classes from {.class} syntax are passed through - Images are NOT clickable by default (use image_section for clickable) */ -}} {{- $src := .Destination -}} {{- $alt := .Text -}} {{- $caption := .Title -}} {{- $class := "" -}} {{- /* Handle class attribute from markdown {.class} syntax */ -}} {{- with .Attributes.class }}{{ $class = . }}{{ end -}} {{- partial "image" (dict "src" $src "page" .Page "alt" $alt "caption" $caption "clickable" false "class" $class ) -}}
